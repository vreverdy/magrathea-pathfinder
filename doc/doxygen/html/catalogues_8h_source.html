248  caract_source[7] = npart_halos[i];
249  caractVect_source.push_back(caract_source);
250  }
251  }
252  else{ // if particles
253 #ifdef VERBOSE
254  if (rank == 0){
255  std::cout<<"# Targets : Particles"<<std::endl;
256  std::cout<<"# Number of particles wanted : "<<parameters.npart<<" "<<" Total number of particles : "<<nparttot<<" Fraction "<<fraction <<" nfiles : "<<partlist.size()<<std::endl;
257  }
258 #endif
259  for(uint ifiling = 0; ifiling < partlist.size(); ++ifiling){
260 #ifdef VERBOSE
261  if(rank == 0)
262  std::cout<<"# File used : "<<partlist[ifiling]<<std::endl;
263 #endif
264  std::vector<float> pos_part, vel_part;
265  std::vector<double> id_part;
266  // Fill vectors from HDF5 files with DM particles properties (we take a random fraction given by 'fraction')
267  TReadHDF5::fillVectors_part(fraction, partlist[ifiling], "data", "position_part", pos_part, "velocity_part", vel_part, "identity_part_ramses", id_part);
268  for(uint i = 0; i < id_part.size(); i++){
269  for(uint j = 0; j < 3; j++){
270  caract_source[j] = pos_part[3*i+j];
271  caract_source[j+3] = vel_part[3*i+j];
272  }
273  caract_source[6] = id_part[i];
274  caract_source[7] = 1;
275  caractVect_source.push_back(caract_source);
276  }
277  }
278  }

431  Integrator::integrate(trajectory, "radius", distTarget, cosmology, octree, vobs, length, parameters.nsteps);
432 
433 
434  const unsigned int marked = trajectory.size() - 1;
435  firstid = marked - (marked > 0);
436  // Check if the photon reached the source

438  result[0][0] = 42;
439  result[0][1] = 42;
440  result[1][0] = 0;
441  result[1][1] = firstid;
442  return result;

445  double previous = trajectory[firstid].chi();
446  double next = trajectory[firstid+1].chi();
447  double f = (next-distTarget)/(next-previous);

459  // Set plane

462  kiTarget[1] = trajectory[firstid].dydl()*f + trajectory[firstid+1].dydl()*(1-f);
463  kiTarget[2] = trajectory[firstid].dzdl()*f + trajectory[firstid+1].dzdl()*(1-f);
464  } else if (parameters.plane == "normal"){
465  kiTarget = finalPoint;
466  } else if(parameters.plane == "exact"){
467  std::cout<<"# Jacobian 'exact' not yet implemented !"<<std::endl;
468  } else{
469  std::cout<<"# WARNING : with jacobiantype = 'bundle', please choose plane = 'sachs', 'normal' or 'exact'"<<std::endl;
470  std::cout<<"# Error at file "<<__FILE__<<", line : "<<__LINE__<<std::endl;
471  std::terminate();
472  }
473

475  if(parameters.stop_bundle == "redshift"){
476  interpRef = redshifts[1];
477  } else if(parameters.stop_bundle == "a"){
478  interpRef = scale_factor;
479  } else if((parameters.stop_bundle == "t") || (parameters.stop_bundle == "eta")){

481  } else if(parameters.stop_bundle == "lambda"){
482  interpRef = trajectory[firstid].lambda()*f + trajectory[firstid+1].lambda()*(1-f);
483  } else if((parameters.stop_bundle == "r") || (parameters.stop_bundle == "radius")){
484  interpRef = distTarget;
485  } else if(parameters.stop_bundle == "plane"){
486  interpRef = kiTarget[0]*finalPoint[0] + kiTarget[1]*finalPoint[1] + kiTarget[2]*finalPoint[2];
487  } else {
488  std::cout<<"# WARNING : Wrong stop criterion for integration"<<std::endl;
489  std::cout<<"# Error at file "<<__FILE__<<", line : "<<__LINE__<<std::endl;
490  std::terminate();
491  }
492  }
493 
494  // Distance between source and photon at the same comoving radius
495  double dist_sep = std::sqrt(pow(trueTarget[0]-finalPoint[0],2)+pow(trueTarget[1]-finalPoint[1],2)+pow(trueTarget[2]-finalPoint[2],2));
496  result[1][0] = dist_sep/distTarget;
497 
498  // Set angle for re-run rejected sources
499  if(parameters.use_previous_catalogues == 2){
500  result[0][0] = phi;
501  result[0][1] = theta;
502  }
503 
504  // Compute root-finder is we do not reach the desired accuracy
505  if(result[1][0] > parameters.cat_accuracy){
506  if(parameters.use_previous_catalogues == 0){ // run catalogues
507  // Compute jacobian for Newton's method
508  // First few rays using the Identity matrix
509  if(iteration < 3){
510  jacobianinv[0][0] = 1;
511  jacobianinv[0][1] = 0;
512  jacobianinv[1][0] = 0;
513  jacobianinv[1][1] = 1;
514  // After a few iterations, use the infinitesimal matrix
515  } else if( (iteration < 5) | ( (parameters.jacobiantype == "infinitesimal") | (parameters.jacobiantype == "infinitesimal_born") ) ){

519  } else{

521  if(jacobian[0][0] == 42 && jacobian[0][1] == 42){
522  result[0][0] = 42;
523  result[0][1] = 42;
524  result[1][0] = 22;

531  // Set the size of the grid used to try to find the null geodesic
532  const double err = 5*result[1][0]/(iteration+1);
533  // Store the angles found in previous iteration in temporary variables
534  double phi_tmp = phi;
535  double theta_tmp = theta;
536  // Store the distance between source and photon for the previous angles
537  double sep_tmp = result[1][0];
538  std::cout<<std::setprecision(17)<<"# Direction : "<<phi<<" "<<theta<<" result[1][0] : "<<result[1][0]<<" err : "<<err<<std::endl;
539  // Launch rays toward the direction of the grid, with the previous angle as the center, while size and resolution are functions of 'err'
540  for(double i = -err; i <= err; i += 0.2*err){
541  for(double j = -err; j <= err; j += 0.2*err){
542  magrathea::Evolution<Photon<double, 3> > trajectory_tmp;
543  photon = Integrator::launch(observer[0], observer[1], observer[2], phi + i, theta + j);
544  trajectory_tmp.append(photon);
545  // Integrate new ray until the comoving radius of the source
546  Integrator::integrate(trajectory_tmp, "radius", distTarget, cosmology, octree, vobs, length, parameters.nsteps);
547  const double marked2 = trajectory_tmp.size()-1;
548  firstid = marked2 - (marked2 > 0);
549  previous = trajectory_tmp[firstid].chi();
550  next = trajectory_tmp[firstid+1].chi();
551  f = (next-distTarget)/(next-previous);
552  finalPoint[0] = trajectory_tmp[firstid].x()*f + trajectory_tmp[firstid+1].x()*(1-f);
553  finalPoint[1] = trajectory_tmp[firstid].y()*f + trajectory_tmp[firstid+1].y()*(1-f);
554  finalPoint[2] = trajectory_tmp[firstid].z()*f + trajectory_tmp[firstid+1].z()*(1-f);
555  // Get new distance between photon and source at the same comoving distance
556  sep_tmp = std::sqrt(pow(trueTarget[0]-finalPoint[0],2)+pow(trueTarget[1]-finalPoint[1],2)+pow(trueTarget[2]-finalPoint[2],2))/distTarget;
557  // If we find one trajectory that gives us the desired accuracy
558  if(sep_tmp < parameters.cat_accuracy){
559  std::cout<<"Good, stop now "<<i<<" "<<j<<" delta : "<<err<<" separation : "<<sep_tmp<<" best for the moment : "<<result[1][0]<<" must be below : "<<parameters.cat_accuracy<<std::endl;
560  result[1][0] = sep_tmp;
561  phi_tmp = phi + i;
562  theta_tmp = theta + j;
563  trajectory = trajectory_tmp;
564  // Prepare for the real jacobian bundle computation
565  if(parameters.jacobiantype == "bundle"){
566  if(parameters.plane == "sachs"){
567  kiTarget[0] = trajectory[firstid].dxdl()*f + trajectory[firstid+1].dxdl()*(1-f);
568  kiTarget[1] = trajectory[firstid].dydl()*f + trajectory[firstid+1].dydl()*(1-f);
569  kiTarget[2] = trajectory[firstid].dzdl()*f + trajectory[firstid+1].dzdl()*(1-f);
570  } else if (parameters.plane == "normal"){
571  kiTarget = finalPoint;
572  } else if(parameters.plane == "exact"){
573  std::cout<<"# Jacobian 'exact' not yet implemented !"<<std::endl;
574  }else{
575  std::cout<<"WARNING : Wrong plane, please choose 'sachs', 'normal' or 'exact'"<<std::endl;
576  std::cout<<"# Error at file "<<__FILE__<<", line : "<<__LINE__<<std::endl;
577  std::terminate();
578  }
579  // Interpolation
580  if(parameters.stop_bundle == "redshift"){
581  interpRef = redshifts[1];
582  } else if(parameters.stop_bundle == "a"){
583  interpRef = scale_factor;
584  } else if((parameters.stop_bundle == "t") || (parameters.stop_bundle == "eta")){
585  interpRef = trajectory[firstid].t()*f + trajectory[firstid+1].t()*(1-f);
586  } else if(parameters.stop_bundle == "lambda"){
587  interpRef = trajectory[firstid].lambda()*f + trajectory[firstid+1].lambda()*(1-f);
588  } else if((parameters.stop_bundle == "r") || (parameters.stop_bundle == "radius")){
589  interpRef = distTarget;
590  } else if(parameters.stop_bundle == "plane"){
591  interpRef = kiTarget[0]*finalPoint[0] + kiTarget[1]*finalPoint[1] + kiTarget[2]*finalPoint[2];
592  }else {
593  std::cout<<"WARNING : Wrong stop criterion for integration"<<std::endl;
594  std::cout<<"# Error at file "<<__FILE__<<", line : "<<__LINE__<<std::endl;
595  std::terminate();
596  }
597  }
598  // Break second loop
599  break;
600  // If trajectory if better than previously but still not the desired accuracy
601  } else if (sep_tmp < result[1][0]){
602  std::cout<<"Better, continue "<<i<<" "<<j<<" delta : "<<err<<" separation : "<<sep_tmp<<" best for the moment : "<<result[1][0]<<" must be below : "<<parameters.cat_accuracy<<std::endl;
603  result[1][0] = sep_tmp;
604  phi_tmp = phi + i;
605  theta_tmp = theta + j;
606  trajectory = trajectory_tmp;
607  } else{
608  // Break second loop
609  continue;
610  }
611  }
612  // Break first loop
613  if(sep_tmp < parameters.cat_accuracy){ // if good trajectory
614  break;
615  } else{
616  continue;
617  }
618  }
619  // Finalise
620  result[0][0] = phi_tmp;
621  result[0][1] = theta_tmp;
622  // Need to iterate on rejected sources
623  if(result[1][0] > parameters.cat_accuracy)
624  return result;
625  }
626 
627  // Newton's method for root-finder (since it did not yet converge), using a 2d cartesian plane in the plane-parallel approximation
628  if(parameters.use_previous_catalogues == 0){
629  const double cp(std::cos(phi)), sp(std::sin(phi)), ct(std::cos(theta)), st(std::sin(theta));
630  //Initialization plane.
631  Point e1, e2, sep, pos, targeted;
632  e1[0] = -sp;
633  e1[1] = cp;
634  e1[2] = 0;
635  e2[0] = -cp*ct;
636  e2[1] = -sp*ct;
637  e2[2] = st;
638  targeted[0] = distTarget*cp*st;
639  targeted[1] = distTarget*sp*st;
640  targeted[2] = distTarget*ct;
641 
642  for(unsigned int idim = 0; idim < 3; idim++){
643  sep[idim] = finalPoint[idim] -trueTarget[idim];
644  pos[idim] = targeted[idim] -trueTarget[idim];
645  }
646  double sepx = sep[0]*e1[0] + sep[1]*e1[1];
647  double sepy = sep[0]*e2[0] + sep[1]*e2[1] + sep[2]*e2[2];
648  double posx = pos[0]*e1[0] + pos[1]*e1[1];
649  double posy = pos[0]*e2[0] + pos[1]*e2[1] + pos[2]*e2[2];
650  double sepxnew = jacobianinv[0][0]*sepx + jacobianinv[0][1]*sepy;
651  double sepynew = jacobianinv[1][0]*sepx + jacobianinv[1][1]*sepy;
652 
653  double posxnew = posx - sepxnew;
654  double posynew = posy - sepynew;
655  double xnew = trueTarget[0]+(posxnew*e1[0]+posynew*e2[0]);
656  double ynew = trueTarget[1]+(posxnew*e1[1]+posynew*e2[1]);
657  double znew = trueTarget[2]+posynew*e2[2];
658  result[0][0] = std::atan2(ynew,xnew);
659  result[0][1] = std::acos(znew/std::sqrt(xnew*xnew+ynew*ynew+znew*znew));
660  return result;
661  }
662  }
663 
664 
665  // Compute jacobian
666  if(parameters.jacobiantype == "bundle"){
667  jacobian = Lensing::dbetadtheta(parameters, kiTarget, interpRef, observer, phi, theta, distTarget, cosmology, octree, vobs, length);
668  } else if(parameters.jacobiantype == "infinitesimal"){
669  jacobian = Lensing::dbetadtheta_infinitesimal(distTarget, trajectory, octree, length);
670  } else if(parameters.jacobiantype == "infinitesimal_born"){
671  trajectory_born.append(photon);
672  Integrator::integrate<-1>(trajectory_born, "radius", distTarget, cosmology, octree, vobs, length, parameters.nsteps);
673  jacobian = Lensing::dbetadtheta_infinitesimal(distTarget, trajectory_born, octree, length);
674  } else{
675  std::cout<<"# jacobiantype must be 'bundle' or 'infinitesimal'"<<std::endl;
676  std::cout<<"# Error at file "<<__FILE__<<", line : "<<__LINE__<<std::endl;
677  std::terminate();
678  }
679  // Check if the jacobian is correct
680  if( jacobian[0][0] == 42 && jacobian[0][1] == 42){
681  result[0][0] = 42;
682  result[0][1] = 42;
683  result[1][0] = 22;
684  return result;
685  }
686 
687  // Compute redshift perturbations
688  const double potential = trajectory[firstid].phi()*f + trajectory[firstid+1].phi()*(1-f);
689  const double isw = trajectory[firstid].isw()*f + trajectory[firstid+1].isw()*(1-f);
690  // Redshift FLRW
691  redshifts[0] = - 1 + 1./(scale_factor);
692 
693  // Compute conversion factor from Ramses Units to SI
694  const double unit_t = scale_factor*scale_factor*parameters.mpc/(h*std::hecto::num*std::kilo::num);
695  const double unit_l = scale_factor*length*100;
696  const double velocityx = velocity[0]*unit_l*1e-2/(unit_t*c); // cm -> m
697  const double velocityy = velocity[1]*unit_l*1e-2/(unit_t*c); // cm -> m
698  const double velocityz = velocity[2]*unit_l*1e-2/(unit_t*c); // cm -> m
699  // Compute the components for the exact definition of redshift at the observer
700  const double gref = -trajectory.front().a()*c*trajectory[0].dtdl()*(1.+trajectory.front().phi()/c2
701  + (vobs[0]*trajectory[0].dxdl() + vobs[1]*trajectory[0].dydl() + vobs[2]*trajectory[0].dzdl())/(c*trajectory[0].dtdl())
702  + 0.5*(vobs[0]*vobs[0] + vobs[1]*vobs[1] + vobs[2]*vobs[2]));
703  const double dtdl = trajectory[firstid].dtdl()*f + trajectory[firstid+1].dtdl()*(1-f);
704  const double dxdl = trajectory[firstid].dxdl()*f + trajectory[firstid+1].dxdl()*(1-f);
705  const double dydl = trajectory[firstid].dydl()*f + trajectory[firstid+1].dydl()*(1-f);
706  const double dzdl = trajectory[firstid].dzdl()*f + trajectory[firstid+1].dzdl()*(1-f);
707  // Compute the components for the exact definition of redshift at the source
708  const double gKmuUnu = -scale_factor*c*dtdl*(1. + potential/c2
709  + (velocityx*dxdl + velocityy*dydl + velocityz*dzdl)/(c*dtdl)
710  + 0.5*(velocityx*velocityx + velocityy*velocityy + velocityz*velocityz));
711  // Redshift FLRW + Potential + Doppler
712  redshifts[2] = redshifts[1] + (velocityx*dxdl + velocityy*dydl + velocityz*dzdl)/(scale_factor*c*dtdl)
713  - (vobs[0]*trajectory[0].dxdl() + vobs[1]*trajectory[0].dydl() + vobs[2]*trajectory[0].dzdl())/(scale_factor*c*trajectory[0].dtdl());
714  // Redshift FLRW + Potential + Doppler + Transverse Doppler
715  redshifts[3] = redshifts[2] + 0.5*(velocityx*velocityx + velocityy*velocityy + velocityz*velocityz - vobs[0]*vobs[0] - vobs[1]*vobs[1] - vobs[2]*vobs[2])/scale_factor;
716  // Redshift FLRW + Potential + Doppler + Transverse Doppler + ISW/RS
717  redshifts[4] = redshifts[3] + isw/scale_factor;
718  // Compute the exact definition of redshift in GR (at first order)
719  redshifts[5] = gKmuUnu/gref - 1;
720 
721  if(parameters.use_previous_catalogues == 0){
722  result[0][0] = phi;
723  result[0][1] = theta;
724  }
725 
726  return result;
727 }
