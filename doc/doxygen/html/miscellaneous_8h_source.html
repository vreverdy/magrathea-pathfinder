139 void Miscellaneous::Tokenize(const std::string& str, std::vector<std::string>& tokens, const std::string& delimiters){
140  // Skip delimiters at beginning.
141  std::string::size_type lastPos = str.find_first_not_of(delimiters, 0);
142  // Find first "non-delimiter".
143  std::string::size_type pos = str.find_first_of(delimiters, lastPos);
144 
145  while (std::string::npos != pos || std::string::npos != lastPos)
146  {
147  // Found a token, add it to the vector.
148  tokens.push_back(str.substr(lastPos, pos - lastPos));
149  // Skip delimiters. Note the "not_of"
150  lastPos = str.find_first_not_of(delimiters, pos);
151  // Find next "non-delimiter"
152  pos = str.find_first_of(delimiters, lastPos);
153  }

526  std::vector<int> selection(posTargets.size(), -1);
527 
528  // Loop over all targets
529  Utility::parallelize(posTargets.size(), [=, &posTargets, &cone, &cones, &selection](const unsigned int ivec){
530  std::array<double,3> position;
531  double reference(0), length(0), distance(0);
532  bool ok = false;
533  position[0] = posTargets[ivec][0];
534  position[1] = posTargets[ivec][1];
535  position[2] = posTargets[ivec][2];
536  reference = 0;
537  length = 0;
538  // Check if inside cone
539  if (cone.inside(position)) {
540  ok = true;
541  // Compute scalar product of cone base and target direction
542  for (unsigned int idim = 0; idim < 3; ++idim) {
543  length += (cone.base(idim)-cone.vertex(idim))*(position[idim]-cone.vertex(idim));
544  }
545  length /= cone.template pow<2>(cone.length());
546  // Compute the ditance between cone base and target
547  for (unsigned int idim = 0; idim < 3; ++idim) {
548  reference += cone.template pow<2>(position[idim]-(cone.vertex(idim)+(cone.base(idim)-cone.vertex(idim))*length));
549  }
550  // Loop over all the other cones
551  for (unsigned int icone = 0; icone < cones.size(); ++icone) {
552  length = 0;
553  distance = 0;
554  if (cones[icone] != cone) {
555  // Compute scalar product of other cone base and target direction
556  for (unsigned int idim = 0; idim < 3; ++idim) {
557  length += (cones[icone].base(idim)-cone.vertex(idim))*(position[idim]-cones[icone].vertex(idim));
558  }
559  if(!(length < 0)){
560  length /= cones[icone].template pow<2>(cones[icone].length());
561  // Compute the ditance between other cone base and target
562  for (unsigned int idim = 0; idim < 3; ++idim) {
563  distance += cones[icone].template pow<2>(position[idim]-(cones[icone].vertex(idim)+(cones[icone].base(idim)-cones[icone].vertex(idim))*length));
564  }
565  // Check if other cone is closer to pixel
566  if (distance < reference) {
567  ok = false;
568  icone = cones.size();
569  }
570  } // if length
571  } // if
572  } // for icone
573  if(ok){
574  selection[ivec] = ivec;
575  }
576  } // if
577  }); // while
578 
579  // Erase targets which are not inside the cone
580  selection.erase(std::remove(std::begin(selection), std::end(selection), -1), std::end(selection));

605 template < class Parameter, class Cone, typename Type1>
606 void Miscellaneous::fill_particles_vectors(const Parameter& parameters, const Cone& cone, const std::vector< std::string >& shellList, std::vector<Type1>& pos_part, std::vector<Type1>& force_part, std::vector<Type1>& potential_part, std::vector<Type1>& a_part, const double thetay, const double thetaz){
607 
608  unsigned long long int marker1(0), marker2(0);
609 
610 #ifdef VERBOSE
611  std::cout<<"# Now loading the position, force and potential of particles"<<std::endl;
612 #endif
613  Miscellaneous::fullclear_vector(pos_part);
614  Miscellaneous::fullclear_vector(force_part);
615  Miscellaneous::fullclear_vector(potential_part);
616  // Loop over files
617  for(uint ifiling = 0; ifiling < shellList.size(); ifiling++){
618  double aexp(0);
619  // Get scale factor of shell
620  TReadHDF5::getAttribute(shellList[ifiling], "metadata/cone_info", "aexp", aexp);
621  // Get data from file, depending on whether it intersects the cone or not
622  TReadHDF5::fillVectors_part(parameters, shellList[ifiling], thetay, thetaz, cone, "data", "position_part", pos_part, "gravitational_field_part", force_part, "potential_part", potential_part);
623  float unit_l(0), unit_t(0);
624  // Get conversion factor from Ramses Units to SI
625  TReadHDF5::getAttribute(shellList[ifiling], "metadata/ramses_info", "unit_l", unit_l); // comobile
626  TReadHDF5::getAttribute(shellList[ifiling], "metadata/ramses_info", "unit_t", unit_t); // superconformal unit
627  // Factors to convert Potential and Force in SI
628  const double factorpot = std::pow(unit_l*1e-2/unit_t, 2);
629  const double factorforce = -aexp*unit_l*1e-2/(unit_t*unit_t);
630  std::transform(std::begin(potential_part)+marker1, std::end(potential_part), std::begin(potential_part)+marker1, std::bind1st(std::multiplies<double>(), factorpot)); // SI units
631  std::transform(std::begin(force_part)+marker2, std::end(force_part), std::begin(force_part)+marker2, std::bind1st(std::multiplies<double>(), factorforce)); // SI units
632  std::vector<Type1> a_tmp(potential_part.size() - marker1);
633  std::fill(a_tmp.begin(), a_tmp.end(), aexp);
634  // Give the same scale factor value to all the particles in the same shell
635  a_part.insert(std::end(a_part), std::begin(a_tmp), std::end(a_tmp));
636  marker1 = potential_part.size();
637  marker2 = force_part.size();
638  }
