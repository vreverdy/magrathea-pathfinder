274  }
275  }

456 template <class Extent, class Data, class Vector, typename Type, unsigned int Dimension, class>
457 inline Data Input::schwarzschildify(const Data& data, const Vector& center, const Vector& position, const Type mass, const Type length)
458 {
459  static const Type extent = Type(Extent::num)/Type(Extent::den);
460  static const Type null = Type();
461  thread_local const Data zero = Data();
462  const Vector v = Utility::join<Dimension>(position, center);
463  const Type r = Utility::distance<Dimension>(position, center);
464  const Type l = length/extent;
465  const Type d = r*l;
466  const Type gm = magrathea::Constants<Type>::g()*mass;
467  Data result = data;
468  result.rho() = data.rho();
469  result.phi() = (d > null) ? (-gm/d) : (std::numeric_limits<Type>::max());
470  result.dphidx() = (d > null) ? ((gm/(d*d))*(v[0]/r)) : (null);
471  result.dphidy() = (d > null) ? ((gm/(d*d))*(v[1]/r)) : (null);
472  result.dphidz() = (d > null) ? ((gm/(d*d))*(v[2]/r)) : (null);
473  result.a() = (result.a() != zero.a()) ? (result.a()) : (!(result.a() != zero.a()));
474  return result;
475 }
