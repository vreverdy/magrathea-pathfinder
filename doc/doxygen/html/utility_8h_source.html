126 template <int Default, typename Type, class Function, class>
127 double Utility::parallelize(const Type nsteps, Function&& function, const int nthreads)
128 {
129  static const Type zero = Type();
130  const std::chrono::high_resolution_clock::time_point tbegin = std::chrono::high_resolution_clock::now();
131  const Type ntasks = std::max(static_cast<int>(1), nthreads);
132  const Type group = std::max(Type(nsteps > zero), nsteps/ntasks);
133  std::vector<std::thread> threads;
134  Type istep = zero;
135  threads.reserve(ntasks);
136  // Each thread takes care of some tasks
137  for (istep = zero; istep < floor(nsteps/ntasks)*ntasks; istep += group) {
138  threads.push_back(std::thread([=, &nsteps, &group, &function](){for (Type i = istep; i < std::min(istep+group, nsteps); ++i) function(i);}));
139  }
140  // Run the last few steps
141  for (istep = istep; istep < nsteps; ++istep) {
142  function(istep);
143  }
144  std::for_each(threads.begin(), threads.end(), [](std::thread& current){current.join();});
145 
146  return std::chrono::duration_cast<std::chrono::duration<double> >(std::chrono::high_resolution_clock::now()-tbegin).count();
147 }

198 template <int Default, typename Iterator, class Function, class>
199 double Utility::parallelize(const Iterator& first, const Iterator& last, Function&& function, const int nthreads)
200 {
201  const std::chrono::high_resolution_clock::time_point tbegin = std::chrono::high_resolution_clock::now();
202  const long long int ntasks = std::max(static_cast<int>(1), nthreads);
203  const long long int group = std::max(static_cast<long long int>(first < last), static_cast<long long int>((last-first)/ntasks));
204  std::vector<std::thread> threads;
205  Iterator it = first;
206  threads.reserve(ntasks);
207  for (it = first; it < last-group; it += group) {
208  threads.push_back(std::thread([=, &last, &group, &function](){std::for_each(it, std::min(it+group, last), function);}));
209  }
210  std::for_each(it, last, function);
211  std::for_each(threads.begin(), threads.end(), [](std::thread& current){current.join();});
212  return std::chrono::duration_cast<std::chrono::duration<double> >(std::chrono::high_resolution_clock::now()-tbegin).count();

481 template <typename Type, class Container, class>
482 inline Type Utility::interpolate(const Type x0, const Container& x, const Container& y, const Container& dydx)
483 {
484  static const Type one = Type(1);
485  const long long int n = std::min(x.size(), y.size());

579  static const std::array<Type, order*2+1> third({{T(-14), T(7), T(13), T(9), T(0), T(-9), T(-13), T(-7), T(14)}});
580  static const std::array<Type, order*2+1> coeff = (Derivative == 0) ? (zeroth) : ((Derivative == 1) ? (first) : ((Derivative == 2) ? (second) : (third)));
581  const long long int n = std::min(x.size(), y.size());
582  const long long int i = std::distance(std::begin(x), std::upper_bound(std::begin(x), std::begin(x)+n, x0, [](const Type x1, const Type x2){return x1 < x2;}));
583  const long long int j = i+((n > 1)*((i <= 0)-(i >= n)));
584  const Type h = (n > 1) ? ((x[j]-x[j-1])*std::max(one, neighbourhood)) : (Type());
585  Type result = Type();
586  if (n > 1) {
587  for (unsigned int k = 0; k < order; ++k) {
588  result += coeff[k]*interpolate(x0-(order-k)*h, x, y)+coeff[order+order-k]*interpolate(x0+(order-k)*h, x, y);
589  }
590  result += coeff[order]*interpolate(x0, x, y);
591  }
592  return result/(normalization[Derivative]*std::pow(h, Type(Derivative)));
593 }

686 template < class Type >
687 inline std::array< std::array<Type, 2>, 2 > Utility::invMatrix2d(const std::array< std::array<Type, 2>, 2 >& A )
688 {

691  result[0][0] = A[1][1]*invdet;
692  result[0][1] = -A[0][1]*invdet;
693  result[1][0] = -A[1][0]*invdet;
694  result[1][1] = A[0][0]*invdet;

703 template < class Type >
704 inline std::array< std::array<Type, 3>, 3 > Utility::invMatrix3d(const std::array< std::array<Type, 3>, 3 >& A )
705 {
706  std::array< std::array<Type, 3>, 3 > result;
707  const double invdet = 1./(A[0][0]*(A[1][1]*A[2][2]-A[2][1]*A[1][2]) - A[0][1]*(A[1][0]*A[2][2]-A[1][2]*A[2][0]) + A[0][2]*(A[1][0]*A[2][1]-A[1][1]*A[2][0]));
708  result[0][0] = (A[1][1]*A[2][2]-A[2][1]*A[1][2])*invdet;
709  result[0][1] = -(A[0][1]*A[2][2]-A[0][2]*A[2][1])*invdet;
710  result[0][2] = (A[0][1]*A[1][2]-A[0][2]*A[1][1])*invdet;
711  result[1][0] = -(A[1][0]*A[2][2]-A[1][2]*A[2][0])*invdet;
712  result[1][1] = (A[0][0]*A[2][2]-A[0][2]*A[2][0])*invdet;
713  result[1][2] = -(A[0][0]*A[1][2]-A[1][0]*A[0][2])*invdet;
714  result[2][0] = (A[1][0]*A[2][1]-A[2][0]*A[1][1])*invdet;
715  result[2][1] = -(A[0][0]*A[2][1]-A[2][0]*A[0][1])*invdet;
716  result[2][2] = (A[0][0]*A[1][1]-A[1][0]*A[0][1])*invdet;
717 
718  return result;
719 }
