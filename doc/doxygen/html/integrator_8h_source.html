186  static const unsigned int y = 1;
187  static const unsigned int z = 2;

189  Vector position = Vector();
190  Scalar length = Scalar();
191  Scalar reference = Scalar();
192  Scalar distance = Scalar();

194  Photon<Type, Dimension> result;
195 
196  // Compute the photon direction
197  do {
198  length = Scalar();
199  reference = Scalar();
200  position = sphere.template random<Dimension-1>(engine, distribution);
201  if (cone.inside(position)) {
202  ok = true;
203  for (unsigned int idim = 0; idim < Dimension; ++idim) {
204  length += (cone.base(idim)-cone.vertex(idim))*(position[idim]-cone.vertex(idim));
205  }
206  length /= cone.template pow<2>(cone.length());

209  }

211  length = Scalar();
212  distance = Scalar();

214  for (unsigned int idim = 0; idim < Dimension; ++idim) {
215  length += (cones[icone].base(idim)-cone.vertex(idim))*(position[idim]-cones[icone].vertex(idim));
216  }
217  if( !(length < 0)){
218  length /= cones[icone].template pow<2>(cones[icone].length());
219  for (unsigned int idim = 0; idim < Dimension; ++idim) {
220  distance += cones[icone].template pow<2>(position[idim]-(cones[icone].vertex(idim)+(cones[icone].base(idim)-cones[icone].vertex(idim))*length));
221  }
222  if (distance < reference) {
223  ok = false;
224  icone = size;
225  }
226  }
227  }
228  }
229  }
230  } while (!ok);

232  // Set the photon cosmology
233  result.a() = one;
234 
235  // Set the photon position

237  result.x() = sphere.center(x);
238  result.y() = sphere.center(y);
239  result.z() = sphere.center(z);

345 template <bool Center, typename Type, unsigned int Dimension, class Container, class>
346 Container Integrator::launch(const Photon<Type, Dimension>& photon, const unsigned int count, const Type angle, const Type rotation)
347 {
348  // Initialization
349  static const unsigned int zero = 0;
350  static const Type one = 1;
351  static const Type four = 4;
352  static const Type pi = four*std::atan(one);
353  const Type step = (pi+pi)/Type(count+(!count));
354  const Type r = std::sqrt((photon.dxdl()*photon.dxdl())+(photon.dydl()*photon.dydl())+(photon.dzdl()*photon.dzdl()));
355  const Type rcos = r*std::cos(angle);
356  const Type rsin = r*std::sin(angle);
357  const Type costheta = std::cos(std::acos(photon.dzdl()/r));
358  const Type sintheta = std::sin(std::acos(photon.dzdl()/r));
359  const Type cosphi = std::cos(std::atan2(photon.dydl(), photon.dxdl()));
360  const Type sinphi = std::sin(std::atan2(photon.dydl(), photon.dxdl()));
361  const Type cospsi = std::cos(rotation);
362  const Type sinpsi = std::sin(rotation);
363  Type x = zero;
364  Type y = zero;
365  Type z = zero;
366  Container result(count+Center, photon);
367 
368  // Loop over photons
369  if (r > zero) {
370  for (unsigned int istep = zero; istep < count; ++istep) {
371  x = rsin*std::cos(istep*step);
372  y = rsin*std::sin(istep*step);
373  z = rcos;
374  result[istep+Center].dxdl() = -cosphi*sinpsi*costheta*x-cosphi*cospsi*costheta*y+cosphi*sintheta*z+sinphi*sinpsi*y-sinphi*cospsi*x;
375  result[istep+Center].dydl() = -sinphi*sinpsi*costheta*x-sinphi*cospsi*costheta*y-cosphi*sinpsi*y+cosphi*cospsi*x+sintheta*sinphi*z;
376  result[istep+Center].dzdl() = sintheta*sinpsi*x+sintheta*cospsi*y+costheta*z;
377  }
378  }
379 
380  // Finalization
381  return result;
382 }

573  data = (Order == 0) ? (octree.ngp(photon.x(), photon.y(), photon.z())) : ((Order == 1) ? (octree.cic(photon.x(), photon.y(), photon.z())) : (Order == 2) ? (octree.tsc(elemsTsc, photon.x(), photon.y(), photon.z())) : (homogeneous));
574  // If photon outside of the Extent, put empty data
575  data = (!(photon.a() < zero) && ((photon.x() > min) && (photon.x() < max) && (photon.y() > min) && (photon.y() < max) && (photon.z() > min) && (photon.z() < max))) ? (data) : (empty);
576  photon.level() = std::get<0>(*octree.locate(photon.x(), photon.y(), photon.z())).level();
577  photon.ah() = data.a();
578  photon.rho() = data.rho();
579  photon.phi() = data.phi();
580  photon.dphidx() = data.dphidx();
581  photon.dphidy() = data.dphidy();
582  photon.dphidz() = data.dphidz();
583  photon.dphidt() = -data.dphidt(); // forward dphidt
584  photon.dphidl() = data.dphidt()*photon.dtdl() + (photon.dphidx()*photon.dxdl() + photon.dphidy()*photon.dydl() + photon.dphidz()*photon.dzdl());
585  photon.laplacian() = (data.phi()-trajectory.back().phi())/dl;
586  photon.dsdl2() = (photon.a()*photon.a())*((-c2*(one+two*(photon.phi()/c2))*photon.dtdl()*photon.dtdl())+((one-two*(photon.phi()/c2))*(photon.dxdl()*photon.dxdl()+photon.dydl()*photon.dydl()+photon.dzdl()*photon.dzdl())));
587  photon.error() = one-((one-two*(photon.phi()/c2))*(photon.dxdl()*photon.dxdl()+photon.dydl()*photon.dydl()+photon.dzdl()*photon.dzdl()))/(c2*(one+two*(photon.phi()/c2))*photon.dtdl()*photon.dtdl());
588  photon.distance() = zero;
589  photon.isw() += (trajectory.back().dphidt()+photon.dphidt())*(photon.t()-trajectory.back().t())/c2;
590  photon.iswold() -= 2*(photon.phi()-trajectory.back().phi())/c2 - ((trajectory.back().dphidx()+photon.dphidx())*(photon.x()-trajectory.back().x())*scale + (trajectory.back().dphidy()+photon.dphidy())*(photon.y()-trajectory.back().y())*scale + (trajectory.back().dphidz()+photon.dphidz())*(photon.z()-trajectory.back().z())*scale)/c2;
591  photon.chi() = std::sqrt(photon.x()*photon.x()+photon.y()*photon.y()+photon.z()*photon.z());
592  photon.lambda() += dl;
593  photon.s() += std::sqrt((photon.x()-trajectory.back().x())*(photon.x()-trajectory.back().x()) + (photon.y()-trajectory.back().y())*(photon.y()-trajectory.back().y()) + (photon.z()-trajectory.back().z())*(photon.z()-trajectory.back().z()));
594 #ifdef VELOCITYFIELD
595  photon.redshift() = a0/photon.a()*(one + phi0c2 - photon.phi()/c2 + photon.isw() + (data.vx()*photon.dxdl() + data.vy()*photon.dydl() + data.vz()*photon.dzdl())/(c*photon.dtdl()) - v0n)-one;
596  //photon.redshift() = a0/photon.a()*(one + (data.vx()*photon.dxdl() + data.vy()*photon.dydl() + data.vz()*photon.dzdl())/(c*photon.dtdl()) - v0n)-one;
597 #else
598  photon.redshift() = a0/photon.a()*(one + phi0c2 - photon.phi()/c2)-one;
599 #endif
600  // Next step
601  // Continue integrating if the photon in still in the Octree
602  if (data != empty) {
603  ratio = photon.a()*photon.a()*(scale/c)/nsteps;
604  dl = std::get<0>(*(octree.locate(photon.x(), photon.y(), photon.z()))).template extent<Type, Position, Extent>()*ratio;
605  trajectory.append(photon);
606  }
607  }
608  // Erase last element if non compatible
609  if (!trajectory.empty()) {
610  if (std::signbit(trajectory.back().redshift()) || (std::signbit(trajectory.back().a()))) {
611  trajectory.pop();
612  }
613  }
614  }
615 
616  // Finalization
617  return trajectory;
618 }
