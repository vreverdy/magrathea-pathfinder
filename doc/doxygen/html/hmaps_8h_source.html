201 void Hmaps::getPixels_per_cone(const Parameter& parameters, const Integer npix, std::vector<Integer>& pixel, std::vector<unsigned int>& ntrajectories, const unsigned int iconerank, const Cones& cones)
202 {
203  // Fill temporary pixel vector with -1
204  std::vector<long> pixeltmp(npix, -1);
205  ntrajectories.push_back(0);
206  // Loop over all the pixels
207  Utility::parallelize(npix, [=, &parameters, &iconerank, &cones, &ntrajectories, &pixeltmp](long pix){
208  double ang1(0), ang2(0), vec[3];
209  bool closest = false;
210  // Convert pixel to 3D vector
211  pix2vec_ring(parameters.nside, pix, vec);
212  // Get the angle between cone base and pixel
213  ang1 = cones[iconerank].direction(0) * vec[0];
214  ang1 += cones[iconerank].direction(1) * vec[1];
215  ang1 += cones[iconerank].direction(2) * vec[2];
216  ang1 = std::acos(ang1 / cones[iconerank].length());
217  // If the angle is smaller than the cone width
218  if (ang1 <= cones[iconerank].angle()){
219  closest = true;
220  // Loop over all the other cones to find if there is another one that is closer
221  for (uint icone = zero; icone < parameters.ncones; icone++){
222  if (icone != static_cast<uint>(iconerank)){
223  // Compute angle for other cone
224  ang2 = cones[icone].direction(0) * vec[0];
225  ang2 += cones[icone].direction(1) * vec[1];
226  ang2 += cones[icone].direction(2) * vec[2];
227  ang2 = std::acos(ang2 / cones[icone].length());
228  // If other cone is closer, then the pixel is not attributed
229  if (ang2 < ang1 && ang2 < cones[icone].angle()){
230  closest = false;
231  break;
232  }
233  }
234  }
235  }
236  if (closest){
237  pixeltmp[pix] = pix;
238  }
239  });
240  // Erase pixels which are not inside the cone
241  pixeltmp.erase(std::remove(std::begin(pixeltmp), std::end(pixeltmp), -1), std::end(pixeltmp));
242  // Count number of pixels inside the cone
243  ntrajectories.back() = pixeltmp.size();
244  // Insert pixels in the vectors for a single MPI task which will take care of several cones
245  pixel.insert(std::end(pixel), std::begin(pixeltmp), std::end(pixeltmp));
246

269  double vec[3];
270  // Convert pixel to 3D vector
271  pix2vec_ring(parameters.nside, pix, vec);
272  double reference(0), length(0), distance(0);
273  bool ok = false;
274  reference = 0;
275  length = 0;
276  // Check if pixel is inside the cone
277  if (cones[iconerank].inside(vec)) {
278  ok = true;
279  // Compute scalar product of cone base and pixel direction
280  for (unsigned int idim = 0; idim < 3; ++idim) {
281  length += (cones[iconerank].base(idim)-cones[iconerank].vertex(idim))*(vec[idim]-cones[iconerank].vertex(idim));
282  }
283  length /= cones[iconerank].template pow<2>(cones[iconerank].length());
284  // Compute the ditance between cone base and pixel
285  for (unsigned int idim = 0; idim < 3; ++idim) {
286  reference += cones[iconerank].template pow<2>(vec[idim]-(cones[iconerank].vertex(idim)+(cones[iconerank].base(idim)-cones[iconerank].vertex(idim))*length));
287  }
288  // Loop over all the other cones
289  for (unsigned int icone = 0; icone < cones.size(); ++icone) {
290  length = 0;
291  distance = 0;
292  if (icone != iconerank) {
293  // Compute scalar product of other cone base and pixel direction
294  for (unsigned int idim = 0; idim < 3; ++idim) {
295  length += (cones[icone].base(idim)-cones[iconerank].vertex(idim))*(vec[idim]-cones[icone].vertex(idim));
296  }
297  if(!(length < 0)){
298  length /= cones[icone].template pow<2>(cones[icone].length());
299  // Compute the ditance between other cone base and pixel
300  for (unsigned int idim = 0; idim < 3; ++idim) {
301  distance += cones[icone].template pow<2>(vec[idim]-(cones[icone].vertex(idim)+(cones[icone].base(idim)-cones[icone].vertex(idim))*length));
302  }
303  // Check if other cone is closer to pixel
304  if (distance < reference) {
305  ok = false;
306  icone = cones.size();
307  }
308  } // if length
309  } // if
310  } // for icone
311  if(ok){
312  pixeltmp[pix] = pix;
313  }
314  } // if
315  });
316 
317  // Erase pixels which are not inside the cone
318  pixeltmp.erase(std::remove(std::begin(pixeltmp), std::end(pixeltmp), -1), std::end(pixeltmp));
319  // Count number of pixels inside the cone
320  ntrajectories.back() = pixeltmp.size();
321  // Insert pixels in the vectors for a single MPI task which will take care of several cones
322  pixel.insert(std::end(pixel), std::begin(pixeltmp), std::end(pixeltmp));

861 void Hmaps::FillMapPropagate(const Parameter& parameters, const Integer ntrajectory, const Integer firsttrajectory, const Octree& octree, const Point& vobs, Map& map, const Integer nmaps, const Pixel& pixel, const Cosmology& cosmology, const point& observer, const Real length, const std::vector<Real>& z_stop_vec){

864  magrathea::Evolution<Photon<double, 3> > reference;
865  Photon<double, 3> photonref;
866  // Launch a ray toward the x-axis
867  photonref = Integrator::launch(0., 0., 0., 1., 0., 0.);
868  const point vobs0 = {0,0,0}; // No peculiar velocity for homogeneous quantities
869  std::vector< unsigned long int > firstid_ref(parameters.nb_z_maps);

873  homotree.assign(parameters.ncoarse/2., zero);
874  // Compute FLRW ray in the homogeneous octree
875  reference = Integrator::propagate<-1>(photonref, parameters.nbundlemin, parameters.openingmin, real(), parameters.stop_ray, cosmology, homotree, vobs0, length, parameters.nsteps*(1 << (parameters.ncoarse-parameters.ncoarse/2))*2, double());
876

878  for(uint iz = 0; iz < parameters.nb_z_maps; iz++){
879  photonref.a() = one/(one + z_stop_vec[iz]);
880  const unsigned long int marked = std::distance(std::begin(reference), std::upper_bound(std::begin(reference), std::end(reference), photonref, [](const Photon<double, 3>& first, const Photon<double, 3>& second){return first.a() > second.a();}));
881  firstid_ref[iz] = marked - (marked > 0);
882  const double previous = reference[firstid_ref[iz]].a();
883  const double next = reference[firstid_ref[iz]+1].a();
884  f_ref[iz] = (next - photonref.a())/(next - previous);
885  distance_ref[iz] = reference[firstid_ref[iz]].distance()*f_ref[iz] + reference[firstid_ref[iz]+1].distance()*(1 - f_ref[iz]);
886  }
