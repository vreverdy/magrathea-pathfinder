354  vertexCone[2] = conic.vertex(2);

362

369  std::string levelName(memb_name);
370  // Enter in groups which start with 'level'
371  if(levelName.substr(0,5)=="level"){
372  hid_t lvlid = H5Gopen(gid,memb_name,H5P_DEFAULT);
373  H5Gget_info (lvlid, &lvlinfo);
374  if(lvlinfo.nlinks!=0){
375  cubeNumber.push_back(std::vector<std::string>());
376  count.push_back(0);
377  // Loop over levels
378  for (unsigned int i1 = 0; i1 < lvlinfo.nlinks; i1++) {
379  H5Lget_name_by_idx (lvlid, ".", H5_INDEX_NAME, H5_ITER_NATIVE, i1, memb_name, (size_t) MAX_NAME, H5P_DEFAULT);
380  std::string str(memb_name);
381  // Select groups which start with 'cube'
382  if(str.substr(0,4) == "cube"){
383  std::array<double, 3> point111;
384  if( parameters.isfullsky == 1 ){
385  TReadHDF5::pos_from_index_fullsky(str, nctab, dimtab, cubesize, point111);
386  }
387  else{
388  TReadHDF5::pos_from_index_narrow(str, nctab, dimtab, cubesize, thetay, thetaz, point111);
389  }
390  double length(0), distance(0), dist111(0);
391  for (unsigned int idim = 0; idim < 3; ++idim) {
392  length += (baseCone[idim]-vertexCone[idim])*(point111[idim]-vertexCone[idim]);
393  dist111 += pow((point111[idim]-vertexCone[idim]),2);
394  }
395  length = length/normCone;
396  for (unsigned int idim = 0; idim < 3; ++idim) {
397  distance += pow(point111[idim]-(vertexCone[idim]+(baseCone[idim]-vertexCone[idim])*(length/normCone)),2);
398  }
399  // Compute if the cube intersects the cone
400  if ( (std::sqrt(dist111) < halfcubediag + microsphere) || ((std::sqrt(distance) < 2*coarsediag + halfcubediag + length*tanAngle) && (length > 0)) ){
401  cubeNumber[i].push_back(str);
402  hid_t cubeid = H5Gopen(lvlid,memb_name,H5P_DEFAULT);
403  hid_t dsid = H5Dopen(cubeid,"refined_bool",H5P_DEFAULT);
404  hsize_t alloc = H5Dget_storage_size(dsid);
405  count[i] += alloc/(sizeof(Integer));
406  H5Dclose(dsid);
407  H5Gclose(cubeid);
408  } // if selection
409  } // if cube
410  } // i1
411  } // if level
412  H5Gclose(lvlid);
413  }// strcmp
414  } // i
415  for(int i = count.size()-1; i >= 0; --i){
416  // If no cells in any group at some given level, then erase level
417  if(count[i] == 0){
418  count.erase(count.begin()+i);
419  cubeNumber.erase(cubeNumber.begin()+i);
420  }
421  }
422  H5Gclose(gid);
423  H5Fclose(file);
