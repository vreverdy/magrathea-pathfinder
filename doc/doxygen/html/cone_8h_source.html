357 template <class Vector, typename Scalar>
358 template <class OtherVector, class>
359 inline bool Cone<Vector, Scalar>::inside(const OtherVector& point) const
360 {
361  Scalar dot = Scalar();
362  Scalar dist = Scalar();
363  Scalar ref = Scalar();
364  for (unsigned int idim = 0; idim < 3; ++idim) {
365  dot += (base(idim)-vertex(idim))*(point[idim]-vertex(idim));
366  dist += pow<2>(point[idim]-vertex(idim));
367  ref += pow<2>(base(idim)-vertex(idim));
368  }
369  return ((ref > Scalar()) && !(std::acos(dot/(rt<2>(dist*ref))) > angle()) && !(dot > ref) && (dot > Scalar()));
