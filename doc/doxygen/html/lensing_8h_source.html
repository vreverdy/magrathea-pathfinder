433  dataxp1 = octree.tsc(elemsTsc, x + 0.5*dl, y, z);
434  dataym1 = octree.tsc(elemsTsc, x , y - 0.5*dl, z);
435  datayp1 = octree.tsc(elemsTsc, x , y + 0.5*dl, z);

437  datazp1 = octree.tsc(elemsTsc, x , y, z + 0.5*dl);

439  dxyp = (datayp1.dphidx() - dataym1.dphidx())/dl;
440  dxzp = (datazp1.dphidx() - datazm1.dphidx())/dl;
441  dyyp = (datayp1.dphidy() - dataym1.dphidy())/dl;
442  dyzp = (datazp1.dphidy() - datazm1.dphidy())/dl;
443  dzzp = (datazp1.dphidz() - datazm1.dphidz())/dl;

452  a11[j][i] = a11[j][i-1] + gxsx*d1d1p*dr;

469  result[j][1][1] = 1 - 2*a22[j][firstid]/c2*length;
